const {
  validateRegistryInstance,
  validateOptionalDimensions,
  validateOptionalLogger
} = require('lerna-test-reporting').inputValidators;

/**
 * Validation functions for validating public input
 * @module SignalFxReporterInputValidators
 * @private
 */
module.exports = {
  /**
   * Validates the public input for the SignalFxMetricsReporter constructor parameters
   * @param signalFxClient The required valid signal fx client
   * @param registry The required registry instance
   * @param {SignalFxMetricsReporterOptions} options The optional parameters
   */
  validateRequiredSignalFxMetricsReporterParameters: (signalFxClient, registry, options) => {
    module.exports.validateSignalFxClient(signalFxClient);
    validateRegistryInstance(registry);
    validateOptionalDimensions(options.defaultDimensions);
    validateOptionalLogger(options.logger);
  },

  /**
   * Validates that the object supplied for the sfx client at least has a send function
   * @param signalFxClient
   */
  validateSignalFxClient: signalFxClient => {
    if (signalFxClient === undefined) {
      throw new Error('signalFxClient was undefined when it is required');
    }

    if (typeof signalFxClient.send !== 'function' || signalFxClient.length < 1) {
      throw new Error('signalFxClient must implement send(data: any)');
    }
  }
};
